pipeline {
    agent any

    environment {
        DOCKER_PATH = '/usr/local/bin/docker'
        COMPOSE_PATH = '/usr/local/bin/docker-compose'
        COMPOSE_FILE = "${env.WORKSPACE}/docker-compose.yml"
    }

    stages {
        stage('Checkout Code') {
            steps {
                git url: 'https://github.com/Higard/pipeline.git', branch: 'main'
            }
        }

        stage('Check Docker Network') {
            steps {
                script {
                    sh """
                    ${DOCKER_PATH} network ls | grep bestNetwork || ${DOCKER_PATH} network create bestNetwork
                    """
                }
            }
        }

        stage('Build and Run with Docker Compose') {
            steps {
                script {
                    sh """
                    ${COMPOSE_PATH} -f ${COMPOSE_FILE} up -d
                    """
                }
            }
        }
    }

    post {
        always {
            script {
                echo "Cleaning up Docker containers..."
                sh """
                ${COMPOSE_PATH} -f ${COMPOSE_FILE} down
                """
            }
        }
    }
}
